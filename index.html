  heartbeat_class  -> s-maxage=30,  stale-while-revalidate=60

method profile:
  read operations  -> GET
  write operations -> POST|PATCH|DELETE

idempotency behavior:
  eligible methods = POST|PATCH|PUT|DELETE
  key header       = Idempotency-Key
  fingerprint      = sha256(method|route|stable_body)
  replay marker    = Idempotency-Replayed: true
  ttl              = 24h
